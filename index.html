6 Python Plot Example

import numpy as np
import matplotlib.pyplot as plt

x = np.linspace(0,10,200)
y = np.sin(x)*np.exp(-0.1*x)

plt.plot(x,y)
plt.title("Damped Oscillation")
plt.xlabel("x")
plt.ylabel("y")
plt.grid(True)
plt.show()
